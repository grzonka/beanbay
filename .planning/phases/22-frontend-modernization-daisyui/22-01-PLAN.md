**Update Dockerfile** — add CSS builder stage BEFORE the Python builder:

```dockerfile
# Stage 0: CSS Builder (Tailwind standalone — no Node.js)
FROM debian:bookworm-slim AS css-builder
WORKDIR /build

# Download Tailwind standalone CLI + daisyUI plugin files
RUN apt-get update && apt-get install -y --no-install-recommends curl ca-certificates && \
    curl -sLO https://github.com/tailwindlabs/tailwindcss/releases/latest/download/tailwindcss-linux-x64 && \
    chmod +x tailwindcss-linux-x64 && \
    curl -sLO https://github.com/saadeghi/daisyui/releases/latest/download/daisyui.mjs && \
    curl -sLO https://github.com/saadeghi/daisyui/releases/latest/download/daisyui-theme.mjs && \
    apt-get purge -y curl && apt-get autoremove -y && rm -rf /var/lib/apt/lists/*

# Copy templates + input CSS (for class scanning)
COPY app/templates/ ./app/templates/
COPY app/static/css/input.css ./app/static/css/input.css

# Build CSS — place daisyui.mjs alongside input.css for @plugin reference
RUN cp daisyui.mjs daisyui-theme.mjs app/static/css/ && \
    ./tailwindcss-linux-x64 -i app/static/css/input.css -o app/static/css/main.css --minify

# Stage 1: Builder (existing — unchanged)
FROM python:3.11-slim AS builder
WORKDIR /build
# ... keep all existing builder steps unchanged ...

# Stage 2: Runtime (existing — add CSS copy)
FROM python:3.11-slim AS runtime
WORKDIR /app
# ... keep all existing runtime steps ...
# ADD THIS LINE after "COPY app/ ./app/":
COPY --from=css-builder /build/app/static/css/main.css ./app/static/css/main.css
```

The key change: insert `Stage 0: CSS Builder` before the existing Python builder stage, and add the `COPY --from=css-builder` line in the runtime stage. The `COPY app/ ./app/` line already copies the rest of the app; the CSS copy from css-builder overwrites the placeholder.
